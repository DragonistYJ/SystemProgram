int fibo(int x)
{
	if (x==0) {
		return 1;
	} else {
		return fibo(x-1)+fibo(x-2);
	}
}

	if (x>y) {
		printf(x > y);
	} else if (x<y) {
		printf(x < y);
	} else {
		printf(x == y);
	}
	return 0;
}
